<script>
  let selectedRating = 5;

  function renderStarPicker() {
    const picker = document.getElementById('commentStarPicker');
    picker.innerHTML = '';
    for (let i = 1; i <= 5; i++) {
      const star = document.createElement('span');
      star.textContent = i <= selectedRating ? '\u2605' : '\u2606';
      star.style.marginRight = '4px';
      star.onclick = () => { selectedRating = i; renderStarPicker(); };
      picker.appendChild(star);
    }
  }

  async function fetchComments() {
    try {
      const res = await fetch('comments.php?action=list');
      const data = await res.json();
      return data.success ? data.comments : [];
    } catch (e) {
      console.error('Could not load comments:', e);
      return [];
    }
  }

  async function postComment() {
    const name = document.getElementById('commentName').value.trim();
    const text = document.getElementById('commentText').value.trim();
    if (!name || !text) { alert('Please enter your name and a comment.'); return; }

    try {
      const res = await fetch('comments.php?action=add', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name, text, rating: selectedRating })
      });
      const data = await res.json();
      if (!data.success) { alert(data.error || 'Could not post comment.'); return; }
      document.getElementById('commentName').value = '';
      document.getElementById('commentText').value = '';
      selectedRating = 5;
      renderStarPicker();
      renderComments();
    } catch (e) {
      alert('Could not connect to the comment server. This feature only works once the site is live on a PHP-enabled host (like TrueHost), not on GitHub Pages or a local test server.');
    }
  }

  async function deleteComment(id) {
    if (!confirm('Delete this comment?')) return;
    await fetch('comments.php?action=delete', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id })
    });
    renderComments();
  }

  async function editComment(id, oldText) {
    const newText = prompt('Edit your comment:', oldText);
    if (newText === null) return;
    await fetch('comments.php?action=edit', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id, text: newText.trim() })
    });
    renderComments();
  }

  async function toggleCommentLike(id, currentlyLiked) {
    await fetch('comments.php?action=like', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id, delta: currentlyLiked ? -1 : 1 })
    });
    renderComments();
  }

  function toggleCommentReply(id) {
    const box = document.getElementById('reply-box-' + id);
    box.classList.toggle('open');
    if (box.classList.contains('open')) box.querySelector('input').focus();
  }

  async function postCommentReply(id) {
    const box = document.getElementById('reply-box-' + id);
    const input = box.querySelector('input');
    if (!input.value.trim()) return;
    await fetch('comments.php?action=reply', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id, text: input.value.trim() })
    });
    renderComments();
  }

  async function renderComments() {
    const list = document.getElementById('commentsList');
    const comments = await fetchComments();
    list.innerHTML = '';
    if (!comments.length) {
      list.innerHTML = '<p style="text-align:center; color:var(--gray2); grid-column:1/-1;">No comments yet. Be the first to share your experience.</p>';
      return;
    }
    comments.forEach(c => {
      const stars = '\u2605'.repeat(c.rating) + '\u2606'.repeat(5 - c.rating);
      const initials = c.name.split(' ').map(w => w[0]).join('').toUpperCase().slice(0, 2);
      const repliesHTML = (c.replies || []).map(r => '<div class="tcard-posted-reply"><span>' + r + '</span></div>').join('');
      const card = document.createElement('div');
      card.className = 'tcard anim';
      const escapedText = c.text.replace(/'/g, "\\'");
      card.innerHTML =
        '<div class="tcard-top">' +
          '<div class="tcard-avatar">' + initials + '</div>' +
          '<div><div class="tcard-author">' + c.name + '</div></div>' +
          '<div class="tcard-rating">' + stars + '</div>' +
        '</div>' +
        '<p class="tcard-text">' + c.text + '</p>' +
        '<div class="tcard-actions">' +
          '<button class="tcard-like" onclick="toggleCommentLike(\'' + c.id + '\', false)"><i class="far fa-heart"></i> <span>' + c.likes + '</span></button>' +
          '<button class="tcard-reply" onclick="toggleCommentReply(\'' + c.id + '\')"><i class="far fa-comment"></i> Reply</button>' +
          '<button class="tcard-reply" onclick="editComment(\'' + c.id + '\', \'' + escapedText + '\')"><i class="far fa-edit"></i> Edit</button>' +
          '<button class="tcard-reply" onclick="deleteComment(\'' + c.id + '\')"><i class="far fa-trash-alt"></i> Delete</button>' +
        '</div>' +
        repliesHTML +
        '<div class="tcard-reply-box" id="reply-box-' + c.id + '"><input type="text" placeholder="Write a reply..."><button onclick="postCommentReply(\'' + c.id + '\')">Post</button></div>';
      list.appendChild(card);
    });
  }

  document.addEventListener('DOMContentLoaded', () => {
    renderStarPicker();
    renderComments();
  });
</script>
